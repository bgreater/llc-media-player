
saveNotes.aspx
note – Note text entered in by the user
netSessionID – GUID of the user’s session
userID – The user’s unique ID in our database
siteID – The site’s unique ID in our database
presentationID – The presentation’s unique ID in our database

Data returned to player - Response.Write("id=" + rInfo.RelativeID + "&userID=" + userID + "&siteID=" + siteID + "&presentationID=" + presentationID + "&netSessionID=" + netSessionID);

saveSeatTime.aspx
timeID – The unique database ID for this seat time.  If -1 is passed it creates a new seat time record.  If you pass the ID it will update the seat time record.
netSessionID – GUID of the user’s session  
userID – The user’s unique ID in our database
siteID – The site’s unique ID in our database
presentationID – The presentation’s unique ID in our database

Data returned to player - Response.Write("id=" + rInfo.RelativeID + "&userID=" + userID + "&siteID=" + siteID + "&presentationID=" + presentationID + "&timeID = " + timeID + "&netSessionID=" + netSessionID);

addBookmark.aspx
title – Name of the bookmark the user entered.
netSessionID - GUID of the user’s session  
timePoint – Millisecond timestamp of the bookmark
userID – The user’s unique ID in our database
siteID – The site’s unique ID in our database
presentationID – The presentation’s unique ID in our database

Data returned to player - Response.Write("id=" + rInfo.RelativeID + "&title=" + title + "&userID=" + userID + "&siteID=" + siteID + "&presentationID=" + presentationID + "&timePoint=" + timePoint + "&netSessionID=" + netSessionID);
